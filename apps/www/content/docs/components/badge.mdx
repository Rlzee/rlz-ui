# Installation
```bash
npm: npx rlz@latest add badge
pnpm: pnpm dlx rlz@latest add badge
yarn: yarn rlz@latest add badge
bun: bunx --bun rlz@latest add badge
```

# Usage
```tsx
import { Badge } from "@ui/components/badge"
```

```tsx
<Badge variant="primary">Primary Badge</Badge>
```
