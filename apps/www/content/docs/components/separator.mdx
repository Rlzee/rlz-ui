# Installation
```bash
npm: npx rlz@latest add separator
pnpm: pnpm dlx rlz@latest add separator
yarn: yarn rlz@latest add separator
bun: bunx --bun rlz@latest add separator
```

# Usage
```tsx
import { Separator } from "@ui/components/separator"
```

```tsx
<div className="flex items-center space-x-4">
  <span>Left content</span>
  <Separator orientation="vertical" />
  <span>Right content</span>
</div>
```
