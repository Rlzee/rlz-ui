# Installation
```bash
npm: npx rlz@latest add dropdown-menu
pnpm: pnpm dlx rlz@latest add dropdown-menu
yarn: yarn rlz@latest add dropdown-menu
bun: bunx --bun rlz@latest add dropdown-menu
```

# Usage
```tsx
import { DropdownMenu } from "@ui/components/dropdown-menu"
```

```tsx
<DropdownMenu>
  <DropdownMenu.Trigger>
    <Button variant="outline">Open</Button>
  </DropdownMenu.Trigger>
  <DropdownMenu.Content>
    <DropdownMenu.Label>My Account</DropdownMenu.Label>
    <DropdownMenu.Separator />
    <DropdownMenu.Item>Profile</DropdownMenu.Item>
    <DropdownMenu.Item>Billing</DropdownMenu.Item>
    <DropdownMenu.Item>Team</DropdownMenu.Item>
    <DropdownMenu.Item>Subscription</DropdownMenu.Item>
  </DropdownMenu.Content>
</DropdownMenu>
```

```tsx
<DropdownMenu>
  <DropdownMenu.Trigger asChild>
    <Button variant="outline">
      Open Menu
    </Button>
  </DropdownMenu.Trigger>
  <DropdownMenu.Content className="w-56">
    <DropdownMenu.Label>My Account</DropdownMenu.Label>
    <DropdownMenu.Separator />
    <DropdownMenu.Group>
      <DropdownMenu.Item>
        Profile
        <DropdownMenu.Shortcut>⇧⌘P</DropdownMenu.Shortcut>
      </DropdownMenu.Item>
      <DropdownMenu.Item>
        Billing
        <DropdownMenu.Shortcut>⌘B</DropdownMenu.Shortcut>

        Settings
        <DropdownMenu.Shortcut>⌘S</DropdownMenu.Shortcut>
      </DropdownMenu.Item>
    </DropdownMenu.Group>
    <DropdownMenu.Separator />
    <DropdownMenu.Item>Team</DropdownMenu.Item>
    <DropdownMenu.Sub>
      <DropdownMenu.SubTrigger>Invite users</DropdownMenu.SubTrigger>
      <DropdownMenu.SubContent>
        <DropdownMenu.Item>Email</DropdownMenu.Item>
```

```tsx
<DropdownMenu>
```

## DropdownMenu (Main)

A menu opened by a trigger which can contain items, submenus, and more.

| Prop | Type | Description | Default |
|------|------|-------------|---------|
| open | `boolean` | The controlled open state of the dropdown menu. | `undefined` |
| defaultOpen | `boolean` | The default open state when initially rendered. | `false` |
| onOpenChange | `(open: boolean) => void` | Event handler called when the open state changes. | `undefined` |
| modal | `boolean` | Whether the dropdown menu should be modal. | `true` |
| dir | `"ltr"` \| `"rtl"` | The reading direction of the dropdown menu. | `"ltr"` |
| children | `ReactNode` | The trigger and content elements. | `undefined` |

## DropdownMenu.Trigger

The button that toggles the dropdown menu.

| Prop | Type | Description | Default |
|------|------|-------------|---------|
| asChild | `boolean` | Renders the trigger as another component via Slot. | `false` |
| children | `ReactNode` | The trigger content. | `undefined` |
| ...props | `ComponentProps<typeof DropdownMenuPrimitive.Trigger>` | All other props are forwarded to the underlying Radix UI component. | `{}` |

## DropdownMenu.Content

The component that pops out when the dropdown menu is open.

| Prop | Type | Description | Default |
|------|------|-------------|---------|
| sideOffset | `number` | The distance in pixels from the trigger. | `4` |
| side | `"top"` \| `"right"` \| `"bottom"` \| `"left"` | The preferred side of the trigger to render against. | `"bottom"` |
| align | `"start"` \| `"center"` \| `"end"` | The preferred alignment against the trigger. | `"center"` |
| alignOffset | `number` | An offset in pixels from the "start" or "end" alignment options. | `0` |
| avoidCollisions | `boolean` | When true, overrides the side and align preferences to prevent collisions. | `true` |
| collisionBoundary | `Element \| null \| Array<Element \| null>` | The element used as the collision boundary. | `[]` |
| collisionPadding | `number \| Partial<Record<Side, number>>` | The distance in pixels from the boundary edges where collision detection should occur. | `0` |
| arrowPadding | `number` | The padding between the arrow and the edges of the content. | `0` |
| sticky | `"partial"` \| `"always"` | The sticky behavior on the align axis. | `"partial"` |
| hideWhenDetached | `boolean` | Whether to hide the content when the trigger becomes fully occluded. | `false` |
| className | `string` | Additional CSS classes to apply to the content. | Predefined styles |
| children | `ReactNode` | The dropdown menu content. | `undefined` |
| ...props | `ComponentProps<typeof DropdownMenuPrimitive.Content>` | All other props are forwarded to the underlying Radix UI component. | `{}` |

## DropdownMenu.Item

| Prop | Type | Description | Default |
|------|------|-------------|---------|

| variant | `"default"` \| `"destructive"` | The style variant of the item. | `"default"` |
| disabled | `boolean` | When true, prevents the user from interacting with the item. | `false` |
| onSelect | `(event: Event) => void` | Event handler called when the user selects an item. | `undefined` |
| textValue | `string` | The textual value of the item for typeahead purposes. | `undefined` |
| className | `string` | Additional CSS classes to apply to the item. | Predefined styles |
| children | `ReactNode` | The item content. | `undefined` |
| ...props | `ComponentProps<typeof DropdownMenuPrimitive.Item>` | All other props are forwarded to the underlying Radix UI component. | `{}` |
