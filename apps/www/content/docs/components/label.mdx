# Installation
```bash
npm: npx rlz@latest add label
pnpm: pnpm dlx rlz@latest add label
yarn: yarn rlz@latest add label
bun: bunx --bun rlz@latest add label
```

# Usage
```tsx
import { Label } from "@ui/components/label"
```

```tsx
<Label htmlFor="email">Email</Label>
```
