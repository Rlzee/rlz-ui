# Installation
```bash
npm: npx rlz@latest add input
pnpm: pnpm dlx rlz@latest add input
yarn: yarn rlz@latest add input
bun: bunx --bun rlz@latest add input
```

# Usage
```tsx
import { Input } from "@ui/components/input"
```

```tsx
<Input placeholder="Enter your text..." />
```
