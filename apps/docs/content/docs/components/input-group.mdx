## Usage

```tsx
import { Input } from "@/components/ui/input";
import { InputGroup } from "@/components/ui/input-group";
```

```tsx
<InputGroup>
  <Input type="email" placeholder="Email" unstyled />
  <InputGroup.Addon>
    <MailIcon />
  </InputGroup.Addon>
</InputGroup>
```
