## Usage

```tsx
import { Radio } from "@/components/ui/radio-group";
```

### Shorthand

```tsx
<Radio.Group aria-label="Select frameworks" defaultValue={["next"]}>
  <Label>
    <Radio value="next" />
    Next.js
  </Label>
  <Label>
    <Radio value="vite" />
    Vite
  </Label>
  <Label>
    <Radio value="astro" />
    Astro
  </Label>
</Radio.Group>
```

### Compound

```tsx
<Radio.Group aria-label="Select frameworks" defaultValue={["next"]}>
  <Label>
    <Radio.Root value="next">
      <Radio.Indicator />
    </Radio.Root>
    Next.js
  </Label>
  <Label>
    <Radio.Root value="vite">
      <Radio.Indicator />
    </Radio.Root>
    Vite
  </Label>
  <Label>
    <Radio.Root value="astro">
      <Radio.Indicator />
    </Radio.Root>
    Astro
  </Label>
</Radio.Group>
```
