## Usage

### Simple

```tsx
import { Command } from "@/components/ui/command";
```

```tsx
<Command>
  <Command.Input placeholder="Type a command or search..." />
  <Command.Empty>No Results Found.</Command.Empty>
  <Command.List>
    <Command.Group>
      <Command.GroupLabel>Suggestions</Command.GroupLabel>
      <Command.Item>Linear</Command.Item>
      <Command.Item>
        <span className="flex-1">Figma</span>
        <Command.Shortcut>⌘F</Command.Shortcut>
      </Command.Item>
    </Command.Group>
    <Command.Group>
      <Command.GroupLabel>Commands</Command.GroupLabel>
      <Command.Item>Clipboard History</Command.Item>
      <Command.Item>
        <span className="flex-1">Import Extension</span>
        <Command.Shortcut>⌘I</Command.Shortcut>
      </Command.Item>
    </Command.Group>
  </Command.List>
  <Command.Footer>Command Footer</Command.Footer>
</Command>
```
